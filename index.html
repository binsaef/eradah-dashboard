ربط Google Sheets المباشر

كود Apps Script كامل لتوصيل اللوحة بملف الإكسل حياً عبر Google Sheets

طريقة الربط (3 خطوات فقط)

الخطوة 1: ارفع ملف الإكسل إلى Google Drive وافتحه كـ Google Sheets.
الخطوة 2: من القائمة: Extensions → Apps Script → انسخ الكود أدناه.
الخطوة 3: انسخ رابط النشر (Deploy as Web App) وضعه في متغير SHEET_API_URL في ملف dashboard.js.

📋 كود Google Apps Script — انسخه كاملاً
// ============================================================ // ERADAH DASHBOARD — Google Apps Script Backend // انسخ هذا الكود في Apps Script → Deploy as Web App // ============================================================ function doGet(e) { const sheet = SpreadsheetApp.getActiveSpreadsheet(); const output = buildDashboardData(sheet); return ContentService .createTextOutput(JSON.stringify(output)) .setMimeType(ContentService.MimeType.JSON) .setHeader('Access-Control-Allow-Origin', '*'); } function buildDashboardData(ss) { return { kpi: getKPI(ss), employees: getEmployees(ss), beneficiaries:getBeneficiaries(ss), sessions: getSessions(ss), absence: getAbsence(ss), admin: getAdmin(ss), financial: getFinancial(ss), diagnoses: getDiagnoses(ss), generated_at: new Date().toISOString() }; } // ---- KPI Summary ---- function getKPI(ss) { const emp = ss.getSheetByName('الموظفين').getLastRow() - 1; const bene = ss.getSheetByName('المستفيدين').getLastRow() - 1; const sess = ss.getSheetByName('الجلسات').getLastRow() - 1; const abs = ss.getSheetByName('جدول الغياب').getLastRow() - 1; const adm = ss.getSheetByName('الحالة الإدارية').getLastRow() - 1; // Financial total const finData = ss.getSheetByName('التحصيل') .getDataRange().getValues().slice(1); const total = finData.reduce((s,r) => s + (Number(r[4])||0), 0); // Sessions status const sessData = ss.getSheetByName('الجلسات') .getDataRange().getValues().slice(1); const completed = sessData.filter(r => r[8]==='تمت').length; const cancelled = sessData.filter(r => r[8]==='ملغاة').length; // Fees expected const beneData = ss.getSheetByName('المستفيدين') .getDataRange().getValues().slice(1); const fees = beneData.reduce((s,r) => s + (Number(r[11])||0), 0); return { emp, bene, sess, abs, adm, total, completed, cancelled, fees }; } // ---- Employees ---- function getEmployees(ss) { const data = ss.getSheetByName('الموظفين') .getDataRange().getValues().slice(1) .filter(r => r[1]); return { by_center: groupCount(data, 7), by_specialty: groupCount(data, 4), by_type: groupCount(data, 5), by_period: groupCount(data, 8) }; } // ---- Beneficiaries ---- function getBeneficiaries(ss) { const data = ss.getSheetByName('المستفيدين') .getDataRange().getValues().slice(1) .filter(r => r[1]); const today = new Date(); const ageGroups = {'أقل من 6':0,'6-12':0,'12-18':0,'18+':0}; data.forEach(r => { if (r[4] instanceof Date) { const age = Math.floor((today - r[4]) / (365.25*24*3600*1000)); if (age < 6) ageGroups['أقل من 6']++; else if (age < 12) ageGroups['6-12']++; else if (age < 18) ageGroups['12-18']++; else ageGroups['18+']++; } }); return { by_gender: groupCount(data, 3), by_center: groupCount(data, 5), by_program: groupCount(data, 10), by_funding: groupCount(data, 12), by_age: ageGroups }; } // ---- Sessions ---- function getSessions(ss) { const data = ss.getSheetByName('الجلسات') .getDataRange().getValues().slice(1) .filter(r => r[1]); const monthly = {}; data.forEach(r => { if (r[1] instanceof Date) { const key = r[1].getFullYear()+'-' +String(r[1].getMonth()+1).padStart(2,'0'); monthly[key] = (monthly[key]||0)+1; } }); return { by_status: groupCount(data, 8), by_service: groupCount(data, 6), monthly: monthly }; } // ---- Absence ---- function getAbsence(ss) { const data = ss.getSheetByName('جدول الغياب') .getDataRange().getValues().slice(1) .filter(r => r[1]); return { by_who: groupCount(data, 1), by_reason:groupCount(data, 5) }; } // ---- Admin ---- function getAdmin(ss) { const data = ss.getSheetByName('الحالة الإدارية') .getDataRange().getValues().slice(1) .filter(r => r[1]); return { by_status: groupCount(data, 5) }; } // ---- Financial ---- function getFinancial(ss) { const data = ss.getSheetByName('التحصيل') .getDataRange().getValues().slice(1) .filter(r => r[1]); const monthly = {}; data.forEach(r => { if (r[3] instanceof Date && r[4]) { const key = r[3].getFullYear()+'-' +String(r[3].getMonth()+1).padStart(2,'0'); monthly[key] = (monthly[key]||0) + Number(r[4]); } }); return { by_type: groupCount(data, 5), monthly: monthly, total: data.reduce((s,r)=>s+(Number(r[4])||0),0) }; } // ---- Diagnoses ---- function getDiagnoses(ss) { const data = ss.getSheetByName('تشخيص الحالات') .getDataRange().getValues().slice(1) .filter(r => r[1]); return { top10: groupCount(data, 3), by_severity: groupCount(data, 4) }; } // ---- Helper ---- function groupCount(arr, colIdx) { const counter = {}; arr.forEach(r => { const val = String(r[colIdx]||'').trim(); if (val) counter[val] = (counter[val]||0)+1; }); return counter; }
